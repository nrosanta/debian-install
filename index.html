Here's a quick tip to include Request and Response in SOAPUI NG -

def uRL = context.expand( '${#Project#URL}' )

String rawResponse = new String(context.httpResponse.getRawRequestData(), "UTF-8")

String rawRequest = new String(context.httpResponse.getRawResponseData(), "UTF-8")

log.info "\n*****************************************************************************\nTEST CASE NAME: ${context.testCase.name}\n*****************************************************************************\n"

log.info("\nTest Environment EndPoint is: "+uRL+"\n")

if (rawRequest!=null){

log.info "\n*****************************************************************************\nTEST CASE NAME: ${context.testCase.name}\n*****************************************************************************\n"

log.info("\n======================RAW REQUEST========================\n"+rawRequest+"\n\n\n")

}

log.info "\n*****************************************************************************\nTEST CASE NAME: ${context.testCase.name}\n*****************************************************************************\n"

log.info("\n======================RAW RESPONSE==========================\n"+rawResponse+"\n\n\n")
